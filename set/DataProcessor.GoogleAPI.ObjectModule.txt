Функция ХТТПГЕТ(Хост, АдресРесурса, Заголовки=Неопределено) Экспорт
	//Только HTTPS
	ЗащищенноеСоединениеOpenSSL = Новый ЗащищенноеСоединениеOpenSSL(Неопределено, Неопределено);
	
	Попытка
		HTTPСоединение = Новый HTTPСоединение(Хост,,,,,,ЗащищенноеСоединениеOpenSSL);
	Исключение
		Сообщить(ИнформацияОбОшибке());
		Возврат 0;
	КонецПопытки;
	
	Попытка
		Если Заголовки = Неопределено Тогда
			HTTPЗапрос = Новый HTTPЗапрос(АдресРесурса);
		Иначе 
			HTTPЗапрос = Новый HTTPЗапрос(АдресРесурса, Заголовки);
		КонецЕсли;
		Результат = HTTPСоединение.Получить(HTTPЗапрос);
		HTTPСоединение = Неопределено;
	Исключение
		Сообщить(ИнформацияОбОшибке());
		Возврат 0;
	КонецПопытки;	
	    Возврат Результат;

Функция ХТТППОСТ(Хост, АдресРесурса, Заголовки=Неопределено, ТелоКакСтрока=Неопределено) Экспорт
	//Только HTTPS
	ЗащищенноеСоединениеOpenSSL = Новый ЗащищенноеСоединениеOpenSSL(Неопределено, Неопределено);
	
	Попытка
		HTTPСоединение = Новый HTTPСоединение(Хост,,,,,,ЗащищенноеСоединениеOpenSSL);
	Исключение
		Сообщить(ИнформацияОбОшибке());
		Возврат 0;
	КонецПопытки;
	
	Попытка
		Если Заголовки = Неопределено Тогда
			HTTPЗапрос = Новый HTTPЗапрос(АдресРесурса);

		КонецЕсли;

		Сообщить(ИнформацияОбОшибке());
		Возврат 0;
	КонецПопытки;	
	    Возврат Результат;
